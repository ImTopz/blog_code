---
```c++
#include <iostream>
#include <stack>
using namespace std;
int x[1000000];
int index[1000000];
int main()
{
   int i=0;
   while(cin>>x[i]&&cin>>index[i])
   {
       i++;
   }
   for(int j=0;j<i;j++)
   {
    if(index[j]==0)
    {  
        
       if(j==0)
       {
         cout<<"0 0"<<endl;
         return 0;
        }
        return 0;
    }
    if(j==0)
    {
        cout<<x[j]*index[j]<<" "<<index[j]-1;
    }
    else
    {
       cout<<" "<<x[j]*index[j]<<" "<<index[j]-1;
    }
   }
    
}//错误点在于 一开始时候系数不管是多少 只要指数为0 就要输出0 0
```
